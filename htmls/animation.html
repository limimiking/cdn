import numpy as np
from scipy.interpolate import interp1d
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation
from ipywidgets import interact
import ipywidgets as widgets
from IPython import display

## data
ntime, nvalue = (10, 40)
time = np.linspace(0, 1, ntime)
data = np.random.normal(size=(ntime, nvalue))

f_out = interp1d(time, data, axis=0)

ntime_out = 500
new_time = np.linspace(0, 1, ntime_out)
new_data = f_out(new_time)
## data end

fig, ax = plt.subplots()
scat = ax.scatter([],[])
ax.set_xlim(-3,3)
ax.set_ylim(-3,3)
plt.close()
def animate(t):
    # scat.set_data((new_data[t][:20], new_data[t][20:]))
    scatter_data = np.stack((new_data[t][:20], new_data[t][20:])).T
    scat.set_offsets(scatter_data)
    return scat

anim = FuncAnimation(fig, animate, frames=50, interval=20)
